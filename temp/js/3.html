# 组合式继承

将原型链和借用构造函数的技术组合
优点：避免引用值的共享
缺点：每次实例化子类的时候，都会调用一次父类的方法

function Person (name,age){
    this.age = age;
    this.name = name;
    this.friends = [];
    this.sayHello = function(){
        console.log('hello I am '+name);
    }
}

function Student (name, age, grade){
    this.grade = grade;
    Person.call(this,name,age);
    this.study = function(){
        console.log(name+" is studying");
    }
}

Person.prototype.sayNima = function(){
    console.log(this.name + "nima");
}

Student.prototype = new Person();

Student.prototype.sayNima = function(){
    console.log('niba');
}

var p1 = new Person('nima',12);
var s1 = new Student('niba',14,1);
var s2 = new Student('nihao',17,1);
p1.friends.push('nima');
s1.friends.push('niba');
s2.friends.push('Tom');
console.log(s1);
console.log(s2);
console.log(p1);
s1.sayNima();
p1.sayNima();

# 原型式继承

优点写法简洁，缺点引用类型的值都是共享的

var person = {
    name: 'person',
    age: 12
};
var student = Object.create(person,{
    grade:{
        value:"class 1"
    }
});
student.name='student';
console.log(person);
console.log(student);

# 寄生式继承

优点：简洁，缺点：每次执行继承时都要重新生成构造一个函数，对象上的函数不能复用，导致效率低下，这点与构造函数类似。

function createStudent(original){
    var student = Object.create(original);
    student.sayHello = function(){
        console.log('I am student ',this.name);
    };
    student.friends = [];
    return student;
}
var person = {
    name: "nima",
    sayHello: function(){
        console.log('I am ',this.name);
    }
}

var student1 = createStudent(person);
var student2 = createStudent(person);
student1.friends.push('nima');
student2.friends.push('niba');
console.log(person);
console.log(student1);
console.log(student2);
console.log(student1.name);
console.log(person.name);
student.sayHello();
person.sayHello();

# 寄生组合式继承

通过借用构造函数来继承属性，通过原型链的混成形式来继承方法。 这个方法的高效性在于，实现继承的过程中它只调用了一次Person的构造函数，并且避免了在Student的prototype 上创造不必要的多余属性，是较为常用的继承对象的方法。

function inheritPrototype(subType,superType){
    var prototype = Object.create(superType.prototype);
    prototype.constructor = subType;
    subType.prototype = prototype;
}

function Person (name,age){
    this.age = age;
    this.name = name;
    this.sayHello = function(){
        console.log('hello I am '+name);
    }
}

Person.prototype.sayNima = function(){
    console.log(this.name + "nima");
}

function Student (name, age, grade){
    this.grade = grade;
    Person.call(this,name,age);
    this.study = function(){
        console.log(name+" is studying");
    }
}


inheritPrototype(Student,Person);

Student.prototype.sayNima = function(){
    console.log('niba');
}

var p1 = new Person ('nima',13);
console.log(p1);


var s1 = new Student('niba',12,"class 1");

console.log(s1);
